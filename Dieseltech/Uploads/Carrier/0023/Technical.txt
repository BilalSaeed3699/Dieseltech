8. Linq join query example 

                ViewBag.SubCategoryList = (from categories in dbEntities.tblCategories
                                           join subcategories in dbEntities.tblSubCategories
                                        on categories.CategoryId equals subcategories.CategoryId
                                           select new SubcategoryAttributeList
                                           {
                                               CategoryId = subcategories.CategoryId,
                                               CategoryName = categories.Name,
                                               SubcategoryId = subcategories.SubcategoryId,
                                               SubcategoryName = subcategories.Name
                                           }).ToList();
